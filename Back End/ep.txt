let { WebSocketServer } = require("ws");
const Socket = new WebSocketServer({ port: 443 });
Socket.on("connection", (ws) => {
  console.log("connected");
  ws.on("close", () => {
    console.log("bye bye");
  });
  ws.on("message", (data) => {
    Socket.clients.forEach((client) => {
      console.log(`${data}`);
      client.send(`${data}`);
    });
  });
});
